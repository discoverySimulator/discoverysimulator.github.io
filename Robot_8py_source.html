176 def updateOdometry(self):
177 vd = self.getRightLinearSpeed()
178 vg = self.getLeftLinearSpeed()
179
180 if self._environnement.isReal():
181 vd+=random.uniform(-self._environnement.getNoiseStrengh()*vd,self._environnement.getNoiseStrengh()*vd)
182 vg+=random.uniform(-self._environnement.getNoiseStrengh()*vg,self._environnement.getNoiseStrengh()*vg)
183
184 v = (vd + vg) / 2
185 e = self.getDistanceBetweenWheels()
186 d = v * config["real_update_time_step"]*self._acceleration/60
187
188 x=self._odometryPose.getX()
189 y=self._odometryPose.getY()
190
191 if vd != vg and vd!=-vg: # le robot n'avance pas tout droit et ne tourne pas sur place
192 R = e * (vd + vg) / (vd - vg)
193 # calcul des coordonnées du centre du cercle trajectoire
194 x0 = x - R * cos(radians(self._odometryPose.getOrientation()))
195 y0 = y - R * sin(radians(self._odometryPose.getOrientation()))
196
197 # calcul position du robot
198 dTheta = d / R
199 self._odometryPose.rotate(degrees(dTheta))
200 self._odometryPose.move(x0 + R * cos(radians(self._odometryPose.getOrientation())),
201 y0 + R * sin(radians(self._odometryPose.getOrientation())))
202 elif vd==-vg: # robot tourne sur place
203 dd=vd * config["real_update_time_step"]*self._acceleration/60
204 dTheta=atan(dd/e)
205 self._odometryPose.rotate(degrees(dTheta))
206 else: # robot en ligne droite
207 nx=x-d * sin(radians(self._odometryPose.getOrientation()))
208 ny=y+d * cos(radians(self._odometryPose.getOrientation()))
209 self._odometryPose.move(nx,ny)
210
211 if self._odometryCounter == 0:
212 point = Object(Representation(Point(colors['odometry'])))
213 self._odometry.append(point)
214 if self._odometryDrawn:
215 self._environnement.addVirtualObject(self._odometry[-1], self._odometryPose.getX(), self._odometryPose.getY())
216 else:
217 self._odometry[-1].setPose(Pose(self._odometryPose.getX(), self._odometryPose.getY()))
218 self._odometryCounter = (self._odometryCounter + 1) % self.NUMBER_STEPS_BEFORE_REFRESH
