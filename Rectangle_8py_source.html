51 def getLineDecomposition(self) -> List[QLineF]:
52 lines=[]
53 w = self._width / 2
54 h = self._height / 2
55 sign = [(-1, -1), (-1, 1), (1, 1), (1, -1)] # counterclockwise
56 pts = []
57 xo = self._pose.getX() + self._pose.getRotationCenterX()
58 yo = self._pose.getY() + self._pose.getRotationCenterY()
59 for i in range(4):
60 x = self._pose.getX() + sign[i][0] * w
61 y = self._pose.getY() + sign[i][1] * h
62 dx = x - xo
63 dy = y - yo
64 pts.append(Point.computeTransformation(xo,yo,dx,dy,self._pose.getOrientation()))
65 pts.append(pts[0])
66
67 for i in range(4):
68 lines.append(QLineF(pts[i][0],pts[i][1],pts[i+1][0],pts[i+1][1]))
69 return lines
