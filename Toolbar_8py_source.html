52 def updateTimeElapsed(self,sender):
53 time=sender.time()
54 hours=int(time//3600)
55 time-=hours*3600
56 minutes=int(time//60)
57 time-=minutes*60
58 seconds=time
59
60 str=""
61 if hours>0:
62 str+=f"{hours}h"
63 if minutes>0 or hours>0:
64 str+=f"{'0' if minutes<10 and hours>0 else ''}{minutes}min"
65 str+=f"{'0' if seconds<10 and (minutes>0 or hours>0) else ''}{round(seconds,1) if minutes==0 else int(seconds)}s"
66 self._timeElapsed.setText(str)

156 def __inputValueAcceleration(self):
157 text = self._valueAcceleration.text()
158 if text[0]=='x':
159 text=text.strip('x')
160 try:
161 acc=float(text)
162 if acc >= self.ACCELERATION_MIN and acc <= self.ACCELERATION_MAX:
163 self.__acceleration=acc
164 if acc > self.ACCELERATION_MAX:
165 self.__acceleration= self.ACCELERATION_MAX
166 except ValueError:
167 return
168 finally:
169 self.__accelerationChanged()
170
171 def __accelerationChanged(self):
172 self.__acceleration=round(self.__acceleration, 1)
173 self._valueAcceleration.setText(f'x{self.__acceleration}')
174 self._valueAcceleration.clearFocus()
175 self.notifyObservers("accelerationChanged")

177 def __clickedDecreaseAcceleration(self):
178 self.__acceleration-= 0.1 if self.__acceleration <= 1 else 1.0
179 self.__acceleration=max(self.__acceleration, self.ACCELERATION_MIN)
180 self.__accelerationChanged()

182 def __clickedIncreaseAcceleration(self):
183 self.__acceleration += 0.1 if self.__acceleration < 1 else 1.0
184 self.__acceleration=min(self.__acceleration, self.ACCELERATION_MAX)
185 self.__accelerationChanged()
