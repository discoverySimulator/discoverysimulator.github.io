71 def buildTree(self):
72 for obj in self.__environment.getObjects():
73 if type(obj) != Object:
74 if issubclass(type(obj), tuple(self.__itemsShown)) or (isinstance(obj, Robot) and (Actuator in self.__itemsShown or Sensor in self.__itemsShown)):
75 parent=None
76 if not issubclass(type(obj), tuple(self.__itemsShown)) and isinstance(obj, Robot) and (Actuator in self.__itemsShown or Sensor in self.__itemsShown):
77 sensors = [comp for comp in obj.getComponents() if isinstance(comp,Sensor)]
78 if (Sensor in self.__itemsShown and sensors) or (Actuator in self.__itemsShown and len(obj.getComponents()) - len(sensors) != 0):
79 parent = Item(self, obj.getID(), 12, setBold=True)
80 parent.setIcon(0,QIcon(f"{config['ressourcesPath']}/robotDisabled.svg"))
81 self.__visibilityButtons.append(None)
82 else:
83 parent = Item(self, obj.getID(), 12, setBold=True)
84 classname = [item for item in self.__itemsShown if isinstance(obj, item)][0].__name__
85 parent.setIcon(0,QIcon(f"{config['ressourcesPath']}/objects/{classname.lower()}.svg"))
86 self.__visibilityButtons.append(VisibilityButton(obj.isVisible()))
87 self.setItemWidget(parent, 1, self.__visibilityButtons[-1])
88 if parent is not None:
89 self.__mainItems.append(parent)
90 self.__mainObjects.append(obj)
91 self.__allObjects.append(obj)
92 self.__mainItemsAssociatedChildren.append([])
93 self.__childrenButtons.append([])
94 if hasattr(obj,"getComponents"):
95 for comp in obj.getComponents():
96 if issubclass(type(comp), tuple(self.__itemsShown)):
97 child = Item(parent,comp.getID())
98 self.__visibilityButtons.append(VisibilityButton(comp.isVisible()))
99 if comp.getVisibilityLocked():
100 self.__visibilityButtons[-1].lock()
101 self.setItemWidget(child, 1, self.__visibilityButtons[-1])
102 self.__subItems.append(child)
103 self.__mainItemsAssociatedChildren[-1].append(child)
104 self.__childrenButtons[-1].append(self.__visibilityButtons[-1])
105 self.__allObjects.append(comp)
106 parent.addChild(child)
107 classname = [item for item in self.__itemsShown if isinstance(comp, item)][0].__name__
108 child.setIcon(0,QIcon(f"{config['ressourcesPath']}/objects/{classname.lower()}.svg"))
109 for button in self.__visibilityButtons:
110 if button is not None:
